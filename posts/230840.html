                buffCvsWriter.write(",");
            }
        }
    }

    HSSFRow currRow;
    HSSFCell cell;
    LinkedHashMap row;
    Map.Entry propertyEntry;
    int rowIndex = 0;

            if (propertyIterator.hasNext()) {

            }
        }
        if (iterator.hasNext()) {
            cellIndex = 0;
        }
    }
